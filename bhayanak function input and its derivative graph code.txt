#program to ask user to input a function and return numerical derivative function
import numpy as np
import matplotlib.pyplot as pl
def derivateN():
    
 import matplotlib.pyplot as pl
 import numpy as np
 import scipy as sp
 import sympy as smp
 from scipy.misc import derivative

     
 x=smp.symbols('x ',real=True)
 global t

 t=eval(input("ENTER FUNCTION WHOSE DERIVATIVE FUNCTION IS NEEDED :"))  
 f=(t)#just enter the function here whose numerical derivative function you want
 global w1
 w1=smp.diff(f,x)#w1=2*x now
 global i
 i=smp.lambdify(x,t)
 
 p=eval(input("ENTER INITIAL NUMBER :"))
 q=eval(input("ENTER FINAL NUMBER :"))
 r=eval(input("ENTER HOW MANY NUMBERS BETWEEN THEM :"))
 global a
 a=np.linspace(p,q,r)
 print(a)
 return smp.lambdify(x,w1)#now f_1(x)=2*x
 
y=derivateN()
g=(y(a))
o=i(a)
pl.plot(a,o,linestyle='dashdot',color='green')
pl.plot(a,g,linestyle='dotted',color='red')
pl.xlabel("x axis")
pl.ylabel("y axis")
pl.ylim(-30,20)
pl.legend([t,w1])
pl.show()
